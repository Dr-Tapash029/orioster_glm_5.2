   181→   181→function generateLocalId(): string {
   182→   182→  const t = Date.now().toString(36).toUpperCase().slice(-6)
   183→   183→  const r = Math.random().toString(36).toUpperCase().slice(2, 5)
   184→   184→  return `PT-${t}${r}`
   185→   185→}
   186→   186→
   187→   187→function safeParse<T>(str: string | null | undefined, fallback: T): T {
   188→   188→  if (!str) return fallback
   189→   189→  try {
   190→   190→    return JSON.parse(str) as T
   191→   191→  } catch {
   192→   192→    return fallback
   193→   193→  }
   194→   194→}
   195→   195→
   196→   196→function isValidStep(step: number, patient: PatientRecord | null): boolean {
   197→   197→  // Used by parent to decide if a step's data is already present.
   198→   198→  if (!patient) return step === 1
   199→   199→  switch (step) {
   200→   200→    case 1:
   201→   201→      return !!patient.fullName && !!patient.gender && patient.consentGiven
   202→   202→    case 2:
   203→   203→      return !!patient.chiefComplaint
   204→   204→    case 3:
   205→   205→      return !!patient.pastHistory
   206→   206→    case 4:
   207→   207→      return !!patient.ongoingMedications
   208→   208→    case 5:
   209→   209→      // We rely on completedSteps set for vitals (no field on patient)
   210→   210→      return false
   211→   211→    case 6:
   212→   212→      return !!patient.allergies
   213→   213→    case 7:
   214→   214→      // appointments live in their own table; rely on completedSteps
   215→   215→      return false
   216→   216→    case 8:
   217→   217→      return !!patient.localSummary
   218→   218→    case 9:
   219→   219→      return patient.notificationStatus === 'SENT' || patient.notificationStatus === 'QUEUED'
   220→   220→    case 10:
   221→   221→      return patient.status === 'COMPLETED'
   222→   222→    default:
   223→   223→      return false
   224→   224→  }
   225→   225→}

   239→   239→  const [savedAt, setSavedAt] = useState<number | null>(null)
   240→   240→  const [submitting, setSubmitting] = useState(false)
   241→   241→  const [errorBanner, setErrorBanner] = useState<string | null>(null)
   242→   242→
   243→   243→  // ── Restore in-progress patient if wizardPatientId set ──

   245→   245→    if (!user) return
   246→   246→    if (wizardPatientId) {
   247→   247→      fetch(`/api/patients/${wizardPatientId}`)
   248→   248→        .then((r) => r.json())
   249→   249→        .then((data) => {
   250→   250→          if (data.patient) {
   251→   251→            const p = data.patient as PatientRecord
   252→   252→            setPatient(p)
   253→   253→            // Restore completed steps based on data presence
   254→   254→            const completed = new Set<number>()
   255→   255→            for (const s of WIZARD_STEPS) {
   256→   256→              if (isValidStep(s.id, p)) completed.add(s.id)
   257→   257→            }
   258→   258→            // Vitals / appointments: check via relations
   259→   259→            if (data.patient.vitals?.length > 0) {
   260→   260→              completed.add(5)
   261→   261→              setVitals(data.patient.vitals[0])
   262→   262→            }
   263→   263→            if (data.patient.appointments?.length > 0) {
   264→   264→              completed.add(7)
   265→   265→            }
   266→   266→            setCompletedSteps(completed)
   267→   267→            // Resume at first non-completed step
   268→   268→            const next = WIZARD_STEPS.find((s) => !completed.has(s.id))
   269→   269→            setCurrentStep(next ? next.id : 10)
   270→   270→          }
   271→   271→        })
   272→   272→        .catch(() => toast.error('Failed to restore in-progress patient.'))
   273→   273→        .finally(() => setLoading(false))
   274→   274→    } else {
   275→   275→      setLoading(false)
   276→   276→    }
   277→   277→  }, [user, wizardPatientId])
   278→   278→
   279→   279→  // ── Saved indicator fade ──
   280→   280→  useEffect(() => {
   281→   281→    if (!savedAt) return
   282→   282→    const t = setTimeout(() => setSavedAt(null), 3500)
   283→   283→    return () => clearTimeout(t)
   284→   284→  }, [savedAt])
   285→   285→
   286→   286→  const triggerSaved = useCallback(() => {
   287→   287→    setSaving(true)
   288→   288→    setSavedAt(Date.now())
   289→   289→    setTimeout(() => setSaving(false), 600)
   290→   290→  }, [])
   291→   291→
   292→   292→  const handlePatientCreated = useCallback(
   293→   293→    (p: PatientRecord) => {
   294→   294→      setPatient(p)
   295→   295→      setWizardPatientId(p.id)
   296→   296→    },
   297→   297→    [setWizardPatientId]
   298→   298→  )
   299→   299→
   300→   300→  const handlePatientUpdated = useCallback((patch: Partial<PatientRecord>) => {
   301→   301→    setPatient((prev) => (prev ? { ...prev, ...patch } : prev))
   302→   302→  }, [])
   303→   303→
   304→   304→  const handleVitalsRecorded = useCallback((v: VitalsRecord) => {
   305→   305→    setVitals(v)
   306→   306→  }, [])
   307→   307→
   308→   308→  const advance = useCallback(() => {
   309→   309→    setCompletedSteps((prev) => {
   310→   310→      const next = new Set(prev)
   311→   311→      next.add(currentStep)
   312→   312→      return next
   313→   313→    })
   314→   314→    if (currentStep < 10) {
   315→   315→      setCurrentStep((s) => s + 1)
   316→   316→      window.scrollTo({ top: 0, behavior: 'smooth' })
   317→   317→    }
   318→   318→  }, [currentStep])
   319→   319→
   320→   320→  const goBack = useCallback(() => {
   321→   321→    if (currentStep > 1) {
   322→   322→      setCurrentStep((s) => s - 1)
   323→   323→      window.scrollTo({ top: 0, behavior: 'smooth' })
   324→   324→    }
   325→   325→  }, [currentStep])
   326→   326→
   327→   327→  // ── Submit final (Step 10) ──
   328→   328→  const handleFinalSubmit = useCallback(async () => {
   329→   329→    if (!patient) return
   330→   330→    setSubmitting(true)
   331→   331→    try {
   332→   332→      const res = await fetch(`/api/patients/${patient.id}`, {
   333→   333→        method: 'PATCH',
   334→   334→        headers: { 'Content-Type': 'application/json' },
   335→   335→        body: JSON.stringify({
   336→   336→          status: 'COMPLETED',
   337→   337→          syncStatus: 'QUEUED',
   338→   338→        }),
   339→   339→      })
   340→   340→      if (!res.ok) throw new Error('Failed to finalize patient')
   341→   341→      const data = await res.json()
   342→   342→      handlePatientUpdated(data.patient)
   343→   343→      toast.success('Patient intake completed', {
   344→   344→        description: `${patient.fullName} has been queued for sync.`,
   345→   345→      })
   346→   346→      setWizardPatientId(null)
   347→   347→      setTimeout(() => setView('patients'), 700)
   348→   348→    } catch (e) {
   349→   349→      toast.error('Submit failed', { description: (e as Error).message })
   350→   350→    } finally {
   351→   351→      setSubmitting(false)
   352→   352→    }
   353→   353→  }, [patient, handlePatientUpdated, setWizardPatientId, setView])
   354→   354→
   355→   355→  // ── Cancel wizard ──
   356→   356→  const handleCancel = useCallback(() => {
   357→   357→    if (
   358→   358→      patient &&
   359→   359→      !confirm('Discard this patient intake? The draft will be saved locally and you can resume later.')
   360→   360→    ) {
   361→   361→      return
   362→   362→    }
   363→   363→    setWizardPatientId(null)
   364→   364→    setView('patients')
   365→   365→  }, [patient, setWizardPatientId, setView])
   366→   366→
   367→   367→  const stepPropsBase: Omit<StepProps, 'patient'> = useMemo(
   368→   368→    () => ({
   369→   369→      user: user!,
   370→   370→      online,
   371→   371→      onPatientCreated: handlePatientCreated,
   372→   372→      onPatientUpdated: handlePatientUpdated,
   373→   373→      onVitalsRecorded: handleVitalsRecorded,
   374→   374→      onAdvance: advance,
   375→   375→      triggerSaved,
   376→   376→    }),
   377→   377→    [user, online, handlePatientCreated, handlePatientUpdated, handleVitalsRecorded, advance, triggerSaved]
   378→   378→  )
   379→   379→
   380→   380→  if (loading || !user) {
   381→   381→    return (
   382→   382→      <div className="flex items-center justify-center py-20">
   383→   383→        <div className="h-8 w-8 animate-spin rounded-full border-2 border-primary border-t-transparent" />
   384→   384→      </div>
   385→   385→    )
   386→   386→  }
   387→   387→
   388→   388→  const currentStepMeta = WIZARD_STEPS.find((s) => s.id === currentStep)!
   389→   389→  const progressPct = ((currentStep - 1) / (WIZARD_STEPS.length - 1)) * 100
   390→   390→
   391→   391→  return (
   392→   392→    <div className="space-y-4">
   393→   393→      {/* Header */}
   394→   394→      <div className="flex flex-wrap items-end justify-between gap-3">
   395→   395→        <div>
   396→   396→          <h1 className="text-xl font-bold tracking-tight sm:text-2xl">Patient Entry Wizard</h1>
   397→   397→          <p className="mt-0.5 text-sm text-muted-foreground">
   398→   398→            10-step offline-first intake · privacy firewall at Step 8 · AI advisory only
   399→   399→          </p>
   400→   400→        </div>
   401→   401→        <div className="flex items-center gap-2">
   402→   402→          {patient && (
   403→   403→            <Badge variant="outline" className="gap-1.5 font-mono text-[11px]">

   421→   421→          if (id <= currentStep || completedSteps.has(id)) {

   423→   423→            window.scrollTo({ top: 0, behavior: 'smooth' })
   424→   424→          }
   425→   425→        }}
   426→   426→      />
   427→   427→
   428→   428→      <div className="grid gap-4 lg:grid-cols-[220px_1fr]">
   429→   429→        {/* Desktop vertical stepper */}
   430→   430→        <aside className="hidden lg:block">
   431→   431→          <VerticalStepper
   432→   432→            currentStep={currentStep}
   433→   433→            completedSteps={completedSteps}
   434→   434→            onJump={(id) => {
   435→   435→              if (id <= currentStep || completedSteps.has(id)) {
   436→   436→                setCurrentStep(id)
   437→   437→                window.scrollTo({ top: 0, behavior: 'smooth' })
   438→   438→              }
   439→   439→            }}
   440→   440→          />
   441→   441→        </aside>
   442→   442→
   443→   443→        {/* Step content */}
   444→   444→        <div className="space-y-3">
   445→   445→          <GlassPanel variant="strong" className="p-5 sm:p-6">
   446→   446→            {/* Step header */}
   447→   447→            <div className="mb-5 flex items-start gap-3 border-b border-border/40 pb-4">
   448→   448→              <div
   449→   449→                className={cn(
   450→   450→                  'flex h-10 w-10 flex-shrink-0 items-center justify-center rounded-xl text-sm font-bold',
   451→   451→                  currentStep === 8
   452→   452→                    ? 'bg-emerald-500/15 text-emerald-700 dark:text-emerald-300'
   453→   453→                    : currentStep === 9
   454→   454→                      ? 'bg-violet-500/15 text-violet-700 dark:text-violet-300'
   455→   455→                      : 'bg-primary/15 text-primary'
   456→   456→                )}
   457→   457→              >
   458→   458→                {completedSteps.has(currentStep) ? <Check className="h-5 w-5" /> : currentStep}
   459→   459→              </div>
   460→   460→              <div className="min-w-0 flex-1">
   461→   461→                <div className="flex flex-wrap items-center gap-2">

   463→   463→                  {currentStep === 8 && (

   478→   478→            </div>
   479→   479→
   480→   480→            {/* Step body */}
   481→   481→            <AnimatePresence mode="wait" initial={false}>
   482→   482→              <motion.div
   483→   483→                key={currentStep}
   484→   484→                initial={{ opacity: 0, x: 12 }}
   485→   485→                animate={{ opacity: 1, x: 0 }}
   486→   486→                exit={{ opacity: 0, x: -12 }}
   487→   487→                transition={{ duration: 0.22, ease: 'easeOut' }}
   488→   488→              >
   489→   489→                {currentStep === 1 && (
   490→   490→                  <StepGeneral {...stepPropsBase} patient={patient} />
   491→   491→                )}
   492→   492→                {currentStep === 2 && (
   493→   493→                  <StepComplaint {...stepPropsBase} patient={patient} />
   494→   494→                )}
   495→   495→                {currentStep === 3 && (
   496→   496→                  <StepHistory {...stepPropsBase} patient={patient} />

   499→   499→                  <StepMedications {...stepPropsBase} patient={patient} />

   721→   721→function StepGeneral({ patient, user, onPatientCreated, onPatientUpdated, onAdvance, triggerSaved }: StepProps) {
   722→   722→  const [fullName, setFullName] = useState(patient?.fullName ?? '')
   723→   723→  const [gender, setGender] = useState(patient?.gender ?? '')
   724→   724→  const [age, setAge] = useState<string>(patient?.age != null ? String(patient.age) : '')
   725→   725→  const [dateOfBirth, setDateOfBirth] = useState(patient?.dateOfBirth ?? '')
   726→   726→  const [contact, setContact] = useState(patient?.contact ?? '')
   727→   727→  const [address, setAddress] = useState(patient?.address ?? '')
   728→   728→  const [localIdNumber, setLocalIdNumber] = useState(patient?.localIdNumber ?? '')
   729→   729→  const [bloodGroup, setBloodGroup] = useState(patient?.bloodGroup ?? '')
   730→   730→  const [heightCm, setHeightCm] = useState(patient?.heightCm != null ? String(patient.heightCm) : '')
   731→   731→  const [weightKg, setWeightKg] = useState(patient?.weightKg != null ? String(patient.weightKg) : '')
   732→   732→  const [profession, setProfession] = useState(patient?.profession ?? '')
   733→   733→  const [education, setEducation] = useState(patient?.education ?? '')
   734→   734→  const [consentGiven, setConsentGiven] = useState(patient?.consentGiven ?? false)
   735→   735→  const [saving, setSaving] = useState(false)
   736→   736→  const [error, setError] = useState<string | null>(null)
   737→   737→
   738→   738→  const validate = (): string | null => {
   739→   739→    if (!fullName.trim()) return 'Full name is required'
   740→   740→    if (!gender) return 'Gender is required'
   741→   741→    if (!consentGiven) return 'Patient consent is required to proceed'
   742→   742→    if (age && (isNaN(Number(age)) || Number(age) < 0 || Number(age) > 130)) {
   743→   743→      return 'Please enter a valid age (0–130)'
   744→   744→    }
   745→   745→    return null
   746→   746→  }
   747→   747→
   748→   748→  const handleContinue = async () => {
   749→   749→    const err = validate()
   750→   750→    if (err) {
   751→   751→      setError(err)
   752→   752→      toast.error(err)
   753→   753→      return
   754→   754→    }
   755→   755→    setError(null)
   756→   756→    setSaving(true)
   757→   757→    try {
   758→   758→      const payload = {
   759→   759→        localId: patient?.localId ?? generateLocalId(),
   760→   760→        fullName: fullName.trim(),
   761→   761→        gender,
   762→   762→        dateOfBirth: dateOfBirth || '',
   763→   763→        age: age ? Number(age) : null,
   764→   764→        contact: contact || null,
   765→   765→        address: address || null,
   766→   766→        localIdNumber: localIdNumber || null,
   767→   767→        bloodGroup: bloodGroup || null,
   768→   768→        heightCm: heightCm ? Number(heightCm) : null,
   769→   769→        weightKg: weightKg ? Number(weightKg) : null,
   770→   770→        profession: profession || null,
   771→   771→        education: education || null,
   772→   772→        consentGiven,
   773→   773→        status: patient?.status ?? 'DRAFT',
   774→   774→        syncStatus: patient?.syncStatus ?? 'DRAFT',
   775→   775→        createdBy: user.id,
   776→   776→      }
   777→   777→
   778→   778→      if (patient) {
   779→   779→        // PATCH existing
   780→   780→        const res = await fetch(`/api/patients/${patient.id}`, {
   781→   781→          method: 'PATCH',
   782→   782→          headers: { 'Content-Type': 'application/json' },
   783→   783→          body: JSON.stringify(payload),
   784→   784→        })
   785→   785→        if (!res.ok) throw new Error('Failed to save')
   786→   786→        const data = await res.json()
   787→   787→        onPatientUpdated(data.patient)
   788→   788→      } else {
   789→   789→        // POST new
   790→   790→        const res = await fetch('/api/patients', {
   791→   791→          method: 'POST',
   792→   792→          headers: { 'Content-Type': 'application/json' },
   793→   793→          body: JSON.stringify(payload),
   794→   794→        })
   795→   795→        if (!res.ok) throw new Error('Failed to create patient')
   796→   796→        const data = await res.json()
   797→   797→        onPatientCreated(data.patient)
   798→   798→      }
   799→   799→      triggerSaved()
   800→   800→      toast.success('General information saved')
   801→   801→      onAdvance()
   802→   802→    } catch (e) {
   803→   803→      toast.error('Save failed', { description: (e as Error).message })
   804→   804→    } finally {
   805→   805→      setSaving(false)
   806→   806→    }
   807→   807→  }
   808→   808→
   809→   809→  return (
   810→   810→    <div className="space-y-5">
   811→   811→      {error && (
   812→   812→        <div className="flex items-center gap-2 rounded-lg border border-red-500/30 bg-red-500/10 px-3 py-2 text-xs text-red-700 dark:text-red-300">
   813→   813→          <AlertTriangle className="h-3.5 w-3.5" />
   814→   814→          {error}
   815→   815→        </div>
   816→   816→      )}
   817→   817→
   818→   818→      <div className="grid gap-4 sm:grid-cols-2">
   819→   819→        <Field label="Full name" icon={<User className="h-3.5 w-3.5" />} required>
   820→   820→          <Input
   821→   821→            value={fullName}
   822→   822→            onChange={(e) => setFullName(e.target.value)}
   823→   823→            placeholder="e.g. Aisha Rahman"
   824→   824→          />
   825→   825→        </Field>
   826→   826→        <Field label="Gender" required>
   827→   827→          <RadioGroup
   828→   828→            value={gender}
   829→   829→            onValueChange={setGender}
   830→   830→            className="flex gap-4 pt-1.5"
   831→   831→          >
   832→   832→            <div className="flex items-center gap-2">
   833→   833→              <RadioGroupItem value="FEMALE" id="g-f" />
   834→   834→              <Label htmlFor="g-f">Female</Label>
   835→   835→            </div>
   836→   836→            <div className="flex items-center gap-2">
